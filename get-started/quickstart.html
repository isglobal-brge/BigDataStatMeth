Quick Start

Your First Analysis

This guide walks you through basic usage of BigDataStatMeth.

Load the Package

library(BigDataStatMeth)

Basic Matrix Operations

Matrix Multiplication

Perform efficient block-wise matrix multiplication:

# Create sample matrices
X <- matrix(rnorm(10000), nrow=100, ncol=100)
Y <- matrix(rnorm(10000), nrow=100, ncol=100)

# Standard matrix multiplication
result1 <- X %*% Y

# Block matrix multiplication (more efficient for large matrices)
result2 <- bdblockmult(X, Y, blocksize=128, paral=TRUE)

# Results should be identical (within numerical precision)
all.equal(result1, result2)

Cross Products

Compute cross products efficiently:

# X'X (transpose cross product)
XtX <- bdtCrossprod(X, blocksize=128, paral=TRUE)

# XX' (cross product)
XXt <- bdCrossprod(X, blocksize=128, paral=TRUE)

Working with HDF5 Files

For large datasets that don’t fit in memory, use HDF5 format:

# Create a large matrix
big_matrix <- matrix(rnorm(100000), nrow=1000, ncol=100)

# Save to HDF5
bdCreate_hdf5_matrix_file(
  filename = "mydata.h5",
  dataset = "mymatrix",
  data = big_matrix,
  chunk = c(100, 100)
)

# Perform operations directly on HDF5 file
result <- bdblockmult_hdf5(
  file1 = "mydata.h5",
  dataset1 = "mymatrix",
  file2 = "mydata.h5",
  dataset2 = "mymatrix",
  blocksize = 128,
  paral = TRUE
)

Statistical Analysis

Principal Component Analysis

Perform PCA on large datasets:

# PCA with HDF5 backend
pca_result <- bdPCA_hdf5(
  filename = "mydata.h5",
  dataset = "mymatrix",
  k = 10,  # number of components
  blocksize = 128
)

# Access results
loadings <- pca_result$loadings
scores <- pca_result$scores
variance <- pca_result$variance

Parallel Computing

Enable parallel processing for faster computation:

# Most functions support parallel execution
result <- bdblockmult(
  X, Y,
  blocksize = 128,
  paral = TRUE,        # Enable parallelization
  threads = 4          # Number of threads (NULL = auto-detect)
)

Best Practices

  1. Choose appropriate block size: Larger blocks = less overhead, but more memory
  2. Use HDF5 for large data: When data exceeds available RAM
  3. Enable parallelization: Set paral=TRUE for compute-intensive operations
  4. Monitor memory usage: Use pryr::mem_used() to track memory consumption

Common Patterns

Pattern 1: Processing Large Files

# 1. Create HDF5 file
bdCreate_hdf5_matrix_file("large_data.h5", "dataset1", your_matrix)

# 2. Normalize data
bdNormalize_hdf5("large_data.h5", "dataset1", "dataset1_normalized")

# 3. Perform analysis
pca <- bdPCA_hdf5("large_data.h5", "dataset1_normalized", k=20)

# 4. Export results
write.csv(pca$scores, "pca_scores.csv")

Pattern 2: Block-wise Operations

# Process data in manageable chunks
process_in_blocks <- function(data, blocksize=1000) {
  n_blocks <- ceiling(nrow(data) / blocksize)
  results <- list()
  
  for (i in 1:n_blocks) {
    start_row <- (i-1) * blocksize + 1
    end_row <- min(i * blocksize, nrow(data))
    
    block <- data[start_row:end_row, ]
    results[[i]] <- process_block(block)
  }
  
  return(do.call(rbind, results))
}
