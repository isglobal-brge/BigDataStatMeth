PCA Analysis Example

Overview

This example demonstrates a complete principal component analysis (PCA) workflow using BigDataStatMeth, from data preparation to result visualization.

Objectives

  • Load and prepare large-scale data
  • Normalize data using HDF5 backend
  • Compute principal components
  • Visualize and interpret results
  • Export findings

Setup

library(BigDataStatMeth)
library(ggplot2)
library(tidyverse)

# Set working directory
setwd("path/to/your/project")

Step 1: Data Preparation

Create Sample Data

For this example, we’ll simulate genomic data with 10,000 features and 500 samples:

# Simulate data matrix
set.seed(123)
n_features <- 10000
n_samples <- 500

# Create base patterns
pattern1 <- rnorm(n_features, mean=5, sd=1)
pattern2 <- rnorm(n_features, mean=3, sd=1.5)

# Generate samples with variation
data_matrix <- matrix(nrow=n_features, ncol=n_samples)

for (i in 1:n_samples) {
  if (i <= 250) {
    # Group 1: more pattern1
    data_matrix[, i] <- pattern1 + rnorm(n_features, sd=0.5)
  } else {
    # Group 2: more pattern2
    data_matrix[, i] <- pattern2 + rnorm(n_features, sd=0.5)
  }
}

# Add row and column names
rownames(data_matrix) <- paste0("Feature_", 1:n_features)
colnames(data_matrix) <- paste0("Sample_", 1:n_samples)

cat("Data dimensions:", dim(data_matrix), "\n")

Save to HDF5

# Create HDF5 file
bdCreate_hdf5_matrix_file(
  filename = "pca_example.h5",
  dataset = "raw_data",
  data = data_matrix,
  chunk = c(500, 50),  # Optimize for column access
  compression = 6
)

cat("✓ Data saved to HDF5\n")

Step 2: Data Normalization

Center and Scale

# Normalize data (center and scale)
bdNormalize_hdf5(
  filename = "pca_example.h5",
  dataset = "raw_data",
  output_dataset = "normalized_data",
  center = TRUE,
  scale = TRUE,
  blocksize = 500
)

cat("✓ Data normalized\n")

Verify Normalization

# Check means and standard deviations
stats <- bdgetSDandMean_hdf5(
  filename = "pca_example.h5",
  dataset = "normalized_data"
)

cat("Mean of means:", mean(stats$means), "\n")
cat("Mean of SDs:", mean(stats$sds), "\n")

Step 3: Compute PCA

Run PCA Analysis

# Compute first 50 principal components
pca_result <- bdPCA_hdf5(
  filename = "pca_example.h5",
  dataset = "normalized_data",
  k = 50,  # Number of components
  blocksize = 500,
  paral = TRUE,
  threads = 4
)

cat("✓ PCA completed\n")
cat("Components computed:", ncol(pca_result$scores), "\n")

Examine Results

# Extract components
pc_scores <- pca_result$scores
pc_loadings <- pca_result$loadings
explained_variance <- pca_result$explained_variance

# Variance explained
total_var_explained <- sum(explained_variance) / sum(pca_result$total_variance) * 100
cat("Total variance explained:", round(total_var_explained, 2), "%\n")

Step 4: Visualization

Scree Plot

# Prepare data for plotting
scree_data <- data.frame(
  PC = 1:length(explained_variance),
  Variance = explained_variance,
  Cumulative = cumsum(explained_variance) / sum(pca_result$total_variance) * 100
)

# Create scree plot
ggplot(scree_data, aes(x=PC, y=Variance)) +
  geom_line(color="steelblue", size=1) +
  geom_point(color="steelblue", size=2) +
  labs(
    title = "Scree Plot",
    x = "Principal Component",
    y = "Variance Explained"
  ) +
  theme_minimal()

PC Score Plot

# Create data frame with PC scores and groups
plot_data <- data.frame(
  PC1 = pc_scores[, 1],
  PC2 = pc_scores[, 2],
  Group = ifelse(1:n_samples <= 250, "Group 1", "Group 2"),
  Sample = colnames(data_matrix)
)

# Plot first two PCs
ggplot(plot_data, aes(x=PC1, y=PC2, color=Group)) +
  geom_point(size=2, alpha=0.6) +
  labs(
    title = "PCA: First Two Principal Components",
    x = paste0("PC1 (", round(scree_data$Variance[1], 2), "%)"),
    y = paste0("PC2 (", round(scree_data$Variance[2], 2), "%)")
  ) +
  theme_minimal() +
  theme(legend.position = "top")

Biplot

# Get top features contributing to PC1 and PC2
top_features <- 20
pc1_loadings <- sort(abs(pc_loadings[, 1]), decreasing=TRUE)[1:top_features]
pc2_loadings <- sort(abs(pc_loadings[, 2]), decreasing=TRUE)[1:top_features]

top_feature_names <- unique(c(names(pc1_loadings), names(pc2_loadings)))

# Create biplot
loading_data <- data.frame(
  PC1 = pc_loadings[top_feature_names, 1],
  PC2 = pc_loadings[top_feature_names, 2],
  Feature = top_feature_names
)

ggplot() +
  geom_point(data=plot_data, aes(x=PC1, y=PC2, color=Group), 
             alpha=0.3, size=1) +
  geom_segment(data=loading_data, 
               aes(x=0, y=0, xend=PC1*50, yend=PC2*50),
               arrow=arrow(length=unit(0.2, "cm")), 
               color="red", alpha=0.5) +
  geom_text(data=loading_data, 
            aes(x=PC1*50, y=PC2*50, label=Feature),
            size=2, vjust=-0.5, color="darkred") +
  labs(title="PCA Biplot") +
  theme_minimal()

Step 5: Interpretation

Identify Important Components

# Components explaining >5% variance
important_pcs <- which(explained_variance / sum(pca_result$total_variance) * 100 > 5)

cat("Important components (>5% variance):", important_pcs, "\n")

Feature Contribution

# Top features for PC1
pc1_top_features <- sort(abs(pc_loadings[, 1]), decreasing=TRUE)[1:10]

cat("\nTop 10 features contributing to PC1:\n")
print(pc1_top_features)

Step 6: Export Results

Save PC Scores

# Export scores
write.csv(
  pc_scores,
  file = "pca_scores.csv",
  row.names = TRUE
)

# Export loadings
write.csv(
  pc_loadings,
  file = "pca_loadings.csv",
  row.names = TRUE
)

# Export variance explained
write.csv(
  data.frame(
    PC = 1:length(explained_variance),
    Variance = explained_variance,
    Percent = explained_variance / sum(pca_result$total_variance) * 100,
    Cumulative = cumsum(explained_variance) / sum(pca_result$total_variance) * 100
  ),
  file = "pca_variance.csv",
  row.names = FALSE
)

cat("✓ Results exported\n")

Advanced: Batch Processing

Process multiple datasets:

# Multiple HDF5 files
files <- c("dataset1.h5", "dataset2.h5", "dataset3.h5")

pca_results <- lapply(files, function(f) {
  cat("Processing:", f, "\n")
  
  # Normalize
  bdNormalize_hdf5(f, "data", "normalized")
  
  # Run PCA
  pca <- bdPCA_hdf5(f, "normalized", k=20, paral=TRUE)
  
  return(pca)
})

names(pca_results) <- files
