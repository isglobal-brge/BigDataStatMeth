Matrix Operations Example

Overview

This example covers efficient matrix operations using BigDataStatMeth, with focus on performance optimization and memory management.

Objectives

  • Perform basic and advanced matrix operations
  • Compare performance of different approaches
  • Optimize block size and parallelization
  • Handle large matrices with HDF5

Setup

library(BigDataStatMeth)
library(microbenchmark)
library(ggplot2)

set.seed(42)

Basic Matrix Operations

Matrix Multiplication

# Create test matrices
n <- 2000
X <- matrix(rnorm(n^2), nrow=n, ncol=n)
Y <- matrix(rnorm(n^2), nrow=n, ncol=n)

# Standard R multiplication
result_base <- X %*% Y

# BigDataStatMeth block multiplication
result_bd <- bdblockmult(
  X, Y,
  blocksize = 256,
  paral = TRUE,
  threads = 4
)

# Verify results are equivalent
all.equal(result_base, result_bd, tolerance=1e-10)

Cross Products

# X'X (transpose cross product)
XtX_base <- t(X) %*% X
XtX_bd <- bdtCrossprod(X, blocksize=256, paral=TRUE)

all.equal(XtX_base, XtX_bd)

# XX' (cross product)
XXt_base <- X %*% t(X)
XXt_bd <- bdCrossprod(X, blocksize=256, paral=TRUE)

all.equal(XXt_base, XXt_bd)

Performance Benchmarking

Matrix Size Impact

# Test different matrix sizes
sizes <- c(500, 1000, 2000, 3000)
results <- list()

for (size in sizes) {
  cat("Testing size:", size, "\n")
  
  X <- matrix(rnorm(size^2), nrow=size, ncol=size)
  Y <- matrix(rnorm(size^2), nrow=size, ncol=size)
  
  benchmark <- microbenchmark(
    base = X %*% Y,
    block_serial = bdblockmult(X, Y, blocksize=256, paral=FALSE),
    block_parallel = bdblockmult(X, Y, blocksize=256, paral=TRUE, threads=4),
    times = 5
  )
  
  results[[as.character(size)]] <- benchmark
}

# Plot results
plot_data <- do.call(rbind, lapply(names(results), function(size) {
  df <- summary(results[[size]])
  df$size <- as.numeric(size)
  return(df)
}))

ggplot(plot_data, aes(x=size, y=median, color=expr)) +
  geom_line(size=1) +
  geom_point(size=2) +
  scale_y_log10() +
  labs(
    title = "Matrix Multiplication Performance",
    x = "Matrix Size",
    y = "Time (ms, log scale)",
    color = "Method"
  ) +
  theme_minimal()

Block Size Optimization

# Test different block sizes
X <- matrix(rnorm(2000^2), nrow=2000, ncol=2000)
Y <- matrix(rnorm(2000^2), nrow=2000, ncol=2000)

block_sizes <- c(64, 128, 256, 512, 1024)
times <- numeric(length(block_sizes))

for (i in seq_along(block_sizes)) {
  bs <- block_sizes[i]
  
  time <- system.time({
    result <- bdblockmult(X, Y, blocksize=bs, paral=TRUE, threads=4)
  })
  
  times[i] <- time["elapsed"]
  cat("Block size:", bs, "- Time:", times[i], "s\n")
}

# Find optimal block size
optimal_idx <- which.min(times)
optimal_blocksize <- block_sizes[optimal_idx]

cat("\nOptimal block size:", optimal_blocksize, "\n")

# Visualize
ggplot(data.frame(blocksize=block_sizes, time=times), 
       aes(x=blocksize, y=time)) +
  geom_line(color="steelblue", size=1) +
  geom_point(size=3, color="steelblue") +
  geom_vline(xintercept=optimal_blocksize, 
             linetype="dashed", color="red") +
  labs(
    title = "Block Size Optimization",
    x = "Block Size",
    y = "Time (seconds)"
  ) +
  theme_minimal()

Memory Management

Monitor Memory Usage

library(pryr)

# Track memory for different operations
mem_before <- mem_used()

result <- bdblockmult(X, Y, blocksize=256, paral=TRUE)

mem_after <- mem_used()
mem_used_operation <- mem_after - mem_before

cat("Memory used:", format(mem_used_operation, units="MB"), "\n")

Memory-Efficient Strategies

# Strategy 1: Use smaller block sizes
small_blocks <- bdblockmult(X, Y, blocksize=128, paral=TRUE)

# Strategy 2: Process in batches
process_large_operation <- function(X, Y, batch_size=500) {
  n_batches <- ceiling(nrow(X) / batch_size)
  results <- list()
  
  for (i in 1:n_batches) {
    start_idx <- (i-1) * batch_size + 1
    end_idx <- min(i * batch_size, nrow(X))
    
    X_batch <- X[start_idx:end_idx, ]
    result_batch <- bdblockmult(X_batch, Y, blocksize=256, paral=TRUE)
    
    results[[i]] <- result_batch
    
    # Force garbage collection
    gc()
  }
  
  return(do.call(rbind, results))
}

Practical Tips

1. Choose Right Function

# For X'X, use specific function (faster)
XtX_specific <- bdtCrossprod(X, blocksize=256, paral=TRUE)

# Instead of general multiplication
# XtX_general <- bdblockmult(t(X), X, ...)  # Slower

2. Reuse Computations

# If you need both X'X and X'Y, compute together
XtX <- bdtCrossprod(X, blocksize=256, paral=TRUE)
XtY <- bdblockmult(t(X), Y, blocksize=256, paral=TRUE)

# More efficient than computing transpose twice

3. Profile Before Optimizing

# Profile to find bottlenecks
Rprof("profile.out")

result <- your_complex_computation()

Rprof(NULL)
summaryRprof("profile.out")

Summary

Key takeaways:

  1. Block multiplication is efficient for large matrices
  2. Optimal block size depends on matrix dimensions and system
  3. Parallelization provides significant speedup for large operations
  4. HDF5 backend enables operations on matrices larger than RAM
  5. Memory management is critical for very large computations
